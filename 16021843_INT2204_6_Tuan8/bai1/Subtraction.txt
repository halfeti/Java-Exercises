package lab08;

/**
 *
 * @author Nguyen Oanh
 */
public class Subtraction extends BinaryExpression{
    Expression left;
    Expression right;
    protected Subtraction (Expression left, Expression right){
        this.left=left;
        this.right=right;
    }
    @Override
    public Expression left() {
        return left;
    }

    @Override
    public Expression right() {
        return right;
    }

    @Override
    public String toString() {
        String a = "";
        return a=a + "(" + left.toString() + "-" + right.toString() + ")";
    }

    @Override
    public int evaluate() {
        return left.evaluate()-right.evaluate();
    }

    
}
